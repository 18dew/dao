Asserter

Assertions for unit testing contracts.  Tests are run with the solUnit unit-testing framework.

All assertions are internal (to make compilation faster) but the _ in front of the function name is still omitted.  This is just for convenience.

The library is meant to be extended by test contracts, e.g.

contract ModAdder {

    function addMod(uint a, uint b, uint modulus) constant returns (uint sum) {
        if (modulus == 0)
            throw;
        return addmod(a, b, modulus);
    }

}

contract SomeTest is Asserter {

    function testAdd() {
        var adder = new ModAdder();
        var sum = adder.addMod(50, 66, 30);
        assertUintsEqual(sum, 26, "addition returned the wrong sum");
    }
}
